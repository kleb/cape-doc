# Print dict in sorted order
def printdict(kw):
    sys.stdout.write("{")
    for j, k in enumerate(sorted(list(kw.keys()))):
        v = kw[k]
        if j > 0:
            sys.stdout.write(", ")
        sys.stdout.write("'%s': %s" % (k, repr(v)))
    sys.stdout.write("}\n")
    sys.stdout.flush()
