from cape.optdict import OptionsDict


def test_dict01_simple():
    # Input dictionary
    d = {

        "x": 24.3,
        "X": [
            24.3,
            14.7
        ],
        "d": {

            "id": None
        }

    assert isinstance(opts["x"], float)
    assert isinstance(opts["X"], list)
    assert isinstance(opts["d"], dict)

    assert opts["d"]["id"] is None
