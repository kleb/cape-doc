        # Import MAT module
        import cape.attdb.ftypes.matfile as matfile
        
        # Read MAT file
        db = matfile.MATFile("bullet.mat")
        
        # Rename a column
        db["MACH"] = db.pop("mach")
        db.cols.remove("mach")
        db.cols.append("MACH")
        
        # Write it
        db.write_mat("bullet1.mat")
        
        # Reread

        if sys.version_info.major == 2:
